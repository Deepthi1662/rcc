// Helper method to build a ServiceActivationDetail object
private ServiceActivationDetail buildServiceActivationDetail(String serviceType, boolean isActive, Integer activeStatusInDays) {
    return ServiceActivationDetail.builder()
            .serviceType(serviceType)
            .isActive(isActive)
            .activeStatusInDays(activeStatusInDays)
            .build();
}

// Map of service types to their corresponding flags and last active days
Map<String, Pair<Boolean, Integer>> serviceDetailsMap = Map.of(
    PHONE_BANKING, new Pair<>(serviceActivationStatus.getPhnBankingRegistrationFlag(), null),
    MOBILE_BANKING, new Pair<>(serviceActivationStatus.getMbBankngRegistrationFlag(), serviceActivationStatus.getMbBankingLastActive()),
    NET_BANKING, new Pair<>(serviceActivationStatus.getNetBankingRegistrationFlag(), serviceActivationStatus.getNetBankingLastActive()),
    POS, new Pair<>(serviceActivationStatus.getPosRegistrationFlag(), serviceActivationStatus.getPosLastActive())
);

// Build service activation details using the serviceDetailsMap
List<ServiceActivationDetail> serviceActivationDetails = serviceDetailsMap.entrySet().stream()
    .map(entry -> buildServiceActivationDetail(entry.getKey(), entry.getValue().getKey(), entry.getValue().getValue()))
    .collect(Collectors.toList());

// Set service activation details to the response builder
customer360ResponseBuilder.serviceActivationDetails(serviceActivationDetails);
