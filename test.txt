  public static String convertToMonthYearFormat(BigDecimal value) {
        // Extract the year and month from the BigDecimal (assuming format yyyyMM)
        int yearMonth = value.intValue();  // Convert BigDecimal to int
        int year = yearMonth / 100;        // Get the year (first 4 digits)
        int month = yearMonth % 100;       // Get the month (last 2 digits)

        // Use Calendar to format the month abbreviation
        Calendar calendar = Calendar.getInstance();
        calendar.set(Calendar.YEAR, year);
        calendar.set(Calendar.MONTH, month - 1);  // Month is 0-based in Calendar
        String monthAbbreviation = new SimpleDateFormat("MMM").format(calendar.getTime());

        // Get the last 2 digits of the year
        String yearShort = String.valueOf(year).substring(2);

        // Return the formatted string in the desired format (e.g., Jul'25)
        return monthAbbreviation + "'" + yearShort;
    }
