public static String getFormattedMonth(String key) {
        DateTimeFormatter formatter = DateTimeFormatter.ofPattern(MMM_YY);

        LocalDate currentDate = LocalDate.now();
        LocalDate lastFinancialYearEnd = LocalDate.of(currentDate.getYear() - 1, 3, 31);  // March of last year

        return switch (key) {
            case CURRENT_MONTH -> currentDate.format(formatter);
            case LAST_MONTH -> currentDate.minusMonths(1).format(formatter);
            case LAST_FINANCIAL_YEAR -> lastFinancialYearEnd.format(formatter);
            case CURRENT_MONTH_LAST_YEAR -> currentDate.minusYears(1).format(formatter);
            case PEAK_IN_LAST_18_MONTHS -> currentDate.minusMonths(6).format(formatter);
            default -> null;
        };
    }
