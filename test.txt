package com.kotak.reliability.entity;

import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.Column;
import javax.persistence.Table;
import java.time.LocalDateTime;

@Entity
@Table(name = "rel.rv_details_relationship")
public class RvDetailsRelationship {

    @Id
    @Column(name = "CRN")
    private String crn;

    @Column(name = "CA_AMB")
    private Double caAmb;

    @Column(name = "SA_AMB")
    private Double saAmb;

    @Column(name = "TD_EOP")
    private Double tdEop;

    @Column(name = "MF_AUM")
    private Double mfAum;

    @Column(name = "MF_OTHERS")
    private Double mfOthers;

    @Column(name = "INSURANCE")
    private Double insurance;

    @Column(name = "SANCTION30")
    private Double sanction30;

    @Column(name = "DEMAT30")
    private Double demat30;

    @Column(name = "FMLY_ID")
    private String fmlyId;

    @Column(name = "FMLY_CA_AMB")
    private Double fmlyCaAmb;

    @Column(name = "FMLY_SA_AMB")
    private Double fmlySaAmb;

    @Column(name = "FMLY_TD_EOP")
    private Double fmlyTdEop;

    @Column(name = "FMLY_MF_AUM")
    private Double fmlyMfAum;

    @Column(name = "FMLY_MF_OTHERS")
    private Double fmlyMfOthers;

    @Column(name = "FMLY_INSURANCE")
    private Double fmlyInsurance;

    @Column(name = "FMLY_SANCTION30")
    private Double fmlySanction30;

    @Column(name = "FMLY_DEMAT30")
    private Double fmlyDemat30;

    @Column(name = "CREATED_BY")
    private String createdBy;

    @Column(name = "CREATED_ON")
    private LocalDateTime createdOn;

    @Column(name = "UPDATED_BY")
    private String updatedBy;

    @Column(name = "UPDATED_ON")
    private LocalDateTime updatedOn;

}
