package com.kotak.reliability.entity;

import javax.persistence.*;
import java.math.BigDecimal;
import java.sql.Timestamp;

@Entity
@Table(name = "sa_details")
public class SaDetails {

    @Id
    @Column(name = "CRN")
    public String crn;

    @Column(name = "NUMBER_OF_SA")
    public Integer numberOfSa;

    @Column(name = "TOTAL_SA_BAL")
    public BigDecimal totalSaBal;

    @Column(name = "ACCOUNT_NUM")
    public String accountNum;

    @Column(name = "PRODUCT_TYPE")
    public String productType;

    @Column(name = "PRODUCT_CODE")
    public String productCode;

    @Column(name = "BRANCH_CODE")
    public String branchCode;

    @Column(name = "LOB_CODE")
    public String lobCode;

    @Column(name = "ACCOUNT_OPEN_DATE")
    public Timestamp accountOpenDate;

    @Column(name = "CURR_MONTH_BAL")
    public BigDecimal currMonthBal;

    @Column(name = "PREV_MONTH_BAL")
    public BigDecimal prevMonthBal;

    @Column(name = "MONTH2_BAL")
    public BigDecimal month2Bal;

    @Column(name = "MONTH3_BAL")
    public BigDecimal month3Bal;

    @Column(name = "FYEND_BAL")
    public BigDecimal fyendBal;

    @Column(name = "LASTFY_BAL")
    public BigDecimal lastFyBal;

    @Column(name = "Q1_AQB")
    public BigDecimal q1Aqb;

    @Column(name = "Q2_AQB")
    public BigDecimal q2Aqb;

    @Column(name = "Q3_AQB")
    public BigDecimal q3Aqb;

    @Column(name = "Q4_AQB")
    public BigDecimal q4Aqb;

    @Column(name = "BCIF_LG")
    public String bcifLg;

    @Column(name = "FINCODLOB")
    public String fincodLob;

    @Column(name = "BCIF_LC")
    public String bcifLc;

    @Column(name = "FLEX_LC")
    public String flexLc;

    @Column(name = "BCIF_RM")
    public String bcifRm;

    @Column(name = "FLEX_LG")
    public String flexLg;

    @Column(name = "CREATED_BY")
    public String createdBy;

    @Column(name = "CREATED_ON", columnDefinition = "TIMESTAMP DEFAULT CURRENT_TIMESTAMP")
    public Timestamp createdOn;

    @Column(name = "UPDATED_BY")
    public String updatedBy;

    @Column(name = "UPDATED_ON", columnDefinition = "TIMESTAMP DEFAULT CURRENT_TIMESTAMP")
    public Timestamp updatedOn;
}
