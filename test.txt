@Query(value = "WITH ca_details_cte AS (" +
            "   SELECT * FROM CaDetails WHERE crn = :crn" +
            "), sa_details_cte AS (" +
            "   SELECT * FROM SaDetails WHERE crn = :crn" +
            ") " +
            "SELECT * " +
            "FROM ca_details_cte ca " +
            "FULL OUTER JOIN sa_details_cte sa ON ca.crn = sa.crn")
    Optional<LiabilitiesDTO> findLiabilitiesDataByCrn(@Param("crn") String crn);
