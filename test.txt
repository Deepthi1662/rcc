)

    ca.*, 
    sa.*
FROM
    ca_details_cte ca
LEFT JOIN
    sa_details_cte sa ON ca.crn = sa.crn
UNION
SELECT
    ca.*, 
    sa.*
FROM
    sa_details_cte sa
LEFT JOIN
    ca_details_cte ca ON ca.crn = sa.crn;
